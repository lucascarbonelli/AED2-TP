\section{Módulo Juego}

Descripción del módulo Juego.

\begin{Interfaz}
  
  \Titulo{Operaciones básicas de juego}
  
  \InterfazFuncion{Pokémons}{\In{e}{jugador}, \In{j}{juego}}{itConj(<pokémon, nat>)}
  [$e \in jugadore(j)$]
  {res = CrearIt(clasificarPokemons(pokémons(e, j)))}
  [$\Theta(1)$]
  [Retorna un iterador al conjunto de tuplas <pokémons, cantidad> del jugador e.]
  [El iterador se invalida si y sólo si se elimina el elemento siguiente del iterador sin utilizar la función \NombreFuncion{EliminarSiguiente}.]

  \InterfazFuncion{Expulsados}{\In{j}{juego}}{conj(jugador)}
  {alias($res$, expulsados(j))}
  [$\Theta(1)$]
  [Retorna el conjunto de jugador expulsados por referencia.]
  [ ?? ]
  
  \InterfazFuncion{PosConPokémons}{\In{j}{juego}}{conj(coor)}
  {alias($res$, posConPokémons(j))}
  [$\Theta(1)$]
  [Retorna el conjunto de coordenadas donde hay pokémons por referencia.]
  [ ?? ]
  
  \InterfazFuncion{PokémonEnPos}{\In{c}{coor}, \In{j}{juego}}{pokémon}
  [$c \in$ posConPokémons(j)]
  {$res$ $=$ pokémonEnPos(c, j))}
  [$\Theta(??)$]
  [Retorna por copia el pokémon que se encuentra en la coordenada c.]

  \InterfazFuncion{CantMovimientosParaCaptura}{\In{c}{coor}, \In{j}{juego}}{nat}
  [$c$ $\in$ posConPokémons(j)]
  {$res$ $=$ cantMovimientosParaCaptura(c, j))}
  [$\Theta(??)$]
  [Retorna por copia la cantidad de movimientos para capturar al pokémon que se encuentra en la coordenada c.]

  \InterfazFuncion{CantMismaEspecie}{\In{p}{pokémon}, \In{cp}{multiconj(pokémon)}}{nat}
  {$res$ $=$ cantMismaEspecie(p, cp)}
  [$\Theta(??))$]
  [Retorna la cantidad de pokémons de la especie p que hay en el conjunto cp.]

  
  \Titulo{Especificación de las operaciones auxiliares utilizadas en la interfaz}
  
  \begin{tad}{Juego Extendido}
    \parskip=0pt
    \tadExtiende{\tadNombre{Juego}}
    
    \tadTitulo{otras operaciones (no exportadas)}
    
    \tadAlinearFunciones{clasificarPokemons}{multiconj(pokémon), pokémon}
    \tadOperacion{clasificarPokemons}{multiconj(pokémon)}{conj(<pokémon, nat>)}{}
    \tadOperacion{eliminar}{multiconj(pokémon), pokémon}{multiconj(pokémon)}{}
    
    \tadAxiomas
    \tadAlinearAxiomas{clasificarPokemons(m)}
    \tadAxioma{clasificarPokemons(m)}{\IF $\emptyset$(m) THEN $\emptyset$ ELSE Ag(<dameUno(m), $\#$dameUno(m)>, \\clasificarPokémons(eliminar(m, dameUno(m)))) FI}
    \tadAxioma{eliminar(m, p)}{\IF $\emptyset$(m) THEN $\emptyset$ ELSE {\IF dameUno(m) = p THEN eliminar(sinUno(m), p) ELSE Ag(dameUno(m), eliminar(sinUno(m), p)) FI} FI}
  \end{tad}
  
  
\end{Interfaz}
